Crear varios:

	for ((i=0; i<20; i++)); do docker run -d -p $((8080+$i)):80 --name contenedor$i nombre1; done
Para pararlos:
	for ((i=0; i<20; i++)); do docker stop contenedor$i; done
Para borrarlos:
	for ((i=0; i<20; i++)); do docker rm contenedor$i; done
